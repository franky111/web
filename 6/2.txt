CREATE TABLE dvd (dvd_id int(11) UNSIGNED auto_increment, title VARCHAR(255), production_year YEAR, PRIMARY KEY(dvd_id));
CREATE TABLE customer (customer_id int(11) UNSIGNED auto_increment, first_name VARCHAR(255), last_name VARCHAR(255), passport_code int(11), registration_date date, PRIMARY KEY(customer_id));
CREATE TABLE offer (offer_id int(11) UNSIGNED auto_increment, dvd_id int(11), customer_id int(11), offer_date date, return_date date, PRIMARY KEY(offer_id));
